```yaml
#arista.clab.yml
---
name: arista-labs

mgmt:
  network: mgmt
  ipv4-subnet: 192.168.100.0/24

topology:
  kinds:
    ceos:
      image: ceos:4.28.10.1M
  nodes:
    eos-01:
      kind: ceos
      mgmt-ipv4: 192.168.100.11
    eos-02:
      kind: ceos
      mgmt-ipv4: 192.168.100.12
  links:
    - endpoints: ["eos-01:eth1", "eos-02:eth1"]
```
